The string library provides functions for string manipulation, pattern matching, and formatting. All functions are available in the `string` table.

## Basic String Operations

<Accordion title="string.len">
Returns the length of a string.

```lua
string.len(s)
```

**Example:**
```lua
print(string.len("hello"))     -- 5
print(string.len(""))           -- 0
print(#"hello")                 -- 5 (equivalent)
```
</Accordion>

<Accordion title="string.sub">
Extracts a substring.

```lua
string.sub(s, i [, j])
```

**Parameters:**
- `i` - Start position (1-based, negative counts from end)
- `j` - End position (default: -1)

**Example:**
```lua
local s = "Hello, World!"
print(string.sub(s, 1, 5))    -- "Hello"
print(string.sub(s, 8))       -- "World!"
print(string.sub(s, -6))      -- "World!"
print(string.sub(s, -6, -2))  -- "World"
```
</Accordion>

<Accordion title="string.reverse">
Reverses a string.

```lua
string.reverse(s)
```

**Example:**
```lua
print(string.reverse("hello"))  -- "olleh"
print(string.reverse("12345"))  -- "54321"
```
</Accordion>

<Accordion title="string.lower">
Converts string to lowercase.

```lua
string.lower(s)
```

**Example:**
```lua
print(string.lower("Hello World"))  -- "hello world"
print(string.lower("ABC123"))       -- "abc123"
```
</Accordion>

<Accordion title="string.upper">
Converts string to uppercase.

```lua
string.upper(s)
```

**Example:**
```lua
print(string.upper("hello"))  -- "HELLO"
print(string.upper("abc123")) -- "ABC123"
```
</Accordion>

<Accordion title="string.rep">
Repeats a string n times.

```lua
string.rep(s, n [, sep])
```

**Parameters:**
- `s` - String to repeat
- `n` - Number of repetitions
- `sep` - Optional separator

**Example:**
```lua
print(string.rep("Ha", 3))         -- "HaHaHa"
print(string.rep("*", 10))         -- "**********"
print(string.rep("Lua", 3, "-"))  -- "Lua-Lua-Lua"
```
</Accordion>

## Character Operations

<Accordion title="string.byte">
Returns numerical codes of characters.

```lua
string.byte(s [, i [, j]])
```

**Example:**
```lua
print(string.byte("ABC"))        -- 65
print(string.byte("ABC", 2))     -- 66
print(string.byte("ABC", 1, 3))  -- 65  66  67
```
</Accordion>

<Accordion title="string.char">
Converts numbers to characters.

```lua
string.char(...)
```

**Example:**
```lua
print(string.char(65, 66, 67))      -- "ABC"
print(string.char(72, 105))         -- "Hi"
print(string.char(33, 63))          -- "!?"
```
</Accordion>

## Pattern Matching

<Accordion title="string.find">
Searches for a pattern in a string.

```lua
string.find(s, pattern [, init [, plain]])
```

**Returns:** start, end, captures... or `nil`

**Example:**
```lua
local s = "Hello, World!"
local start, finish = string.find(s, "World")
print(start, finish)  -- 8  12

local s = "The answer is 42"
local pos = string.find(s, "%d+")
print(pos)  -- 15

-- With captures
local s = "name: John, age: 30"
local _, _, name = string.find(s, "name: (%w+)")
print(name)  -- "John"
```
</Accordion>

<Accordion title="string.match">
Searches for the first match of pattern.

```lua
string.match(s, pattern [, init])
```

**Returns:** Captured strings or whole match

**Example:**
```lua
local s = "Today is 2024-03-15"
local date = string.match(s, "%d%d%d%d%-%d%d%-%d%d")
print(date)  -- "2024-03-15"

local email = "user@example.com"
local user, domain = string.match(email, "(.+)@(.+)")
print(user, domain)  -- "user"  "example.com"
```
</Accordion>

<Accordion title="string.gmatch">
Returns an iterator for all matches.

```lua
string.gmatch(s, pattern)
```

**Example:**
```lua
local s = "Hello World from Lua"
for word in string.gmatch(s, "%a+") do
  print(word)
end
-- Output:
-- Hello
-- World
-- from
-- Lua

local text = "x=10, y=20, z=30"
for key, value in string.gmatch(text, "(%w+)=(%d+)") do
  print(key, value)
end
-- Output:
-- x  10
-- y  20
-- z  30
```
</Accordion>

<Accordion title="string.gsub">
Replaces occurrences of a pattern.

```lua
string.gsub(s, pattern, repl [, n])
```

**Parameters:**
- `pattern` - Pattern to find
- `repl` - Replacement (string, table, or function)
- `n` - Maximum replacements

**Returns:** Modified string, number of substitutions

**Example:**
```lua
local s = "Hello World"
print(string.gsub(s, "World", "Lua"))  -- "Hello Lua", 1

-- Remove all spaces
local s = "a b c d"
print(string.gsub(s, " ", ""))  -- "abcd", 3

-- Using captures
local s = "2024-03-15"
print(string.gsub(s, "(%d+)-(%d+)-(%d+)", "%3/%2/%1"))
-- "15/03/2024", 1

-- Using function
local s = "hello"
print(string.gsub(s, ".", function(c)
  return string.char(string.byte(c) + 1)
end))
-- "ifmmp", 5
```
</Accordion>

## Pattern Classes

- `.` - All characters
- `%a` - Letters
- `%c` - Control characters
- `%d` - Digits
- `%g` - Printable characters except space
- `%l` - Lowercase letters
- `%p` - Punctuation characters
- `%s` - Space characters
- `%u` - Uppercase letters
- `%w` - Alphanumeric characters
- `%x` - Hexadecimal digits
- `%z` - Character with representation 0

**Modifiers:**
- `+` - 1 or more repetitions
- `*` - 0 or more repetitions
- `-` - 0 or more repetitions (non-greedy)
- `?` - Optional (0 or 1)

**Example:**
```lua
-- Extract email
local text = "Contact: john@example.com"
local email = string.match(text, "%w+@%w+%.%w+")
print(email)  -- "john@example.com"

-- Validate phone number
local phone = "555-1234"
if string.match(phone, "^%d%d%d%-%d%d%d%d$") then
  print("Valid phone number")
end
```

## Formatting

<Accordion title="string.format">
Formats a string using printf-style formatting.

```lua
string.format(formatstring, ...)
```

**Format specifiers:**
- `%s` - String
- `%d` or `%i` - Integer
- `%f` - Float
- `%x` - Hexadecimal
- `%o` - Octal
- `%e` - Scientific notation
- `%q` - Quoted string

**Example:**
```lua
print(string.format("Hello, %s!", "World"))  -- "Hello, World!"
print(string.format("%d + %d = %d", 2, 3, 5))  -- "2 + 3 = 5"
print(string.format("%.2f", 3.14159))  -- "3.14"
print(string.format("%05d", 42))  -- "00042"
print(string.format("%x", 255))  -- "ff"
print(string.format("%.2e", 1234.5))  -- "1.23e+03"

-- Width and alignment
print(string.format("|%10s|", "hi"))     -- "|        hi|"
print(string.format("|%-10s|", "hi"))    -- "|hi        |"
print(string.format("|%10.2f|", 3.14))  -- "|      3.14|"
```
</Accordion>

## Packing and Unpacking

<Accordion title="string.pack">
Packs values into a binary string.

```lua
string.pack(fmt, v1, v2, ...)
```

**Example:**
```lua
local packed = string.pack("ii", 10, 20)
print(#packed)  -- 8 (two 4-byte integers)

local data = string.pack("ddd", 1.5, 2.5, 3.5)
```
</Accordion>

<Accordion title="string.unpack">
Unpacks values from a binary string.

```lua
string.unpack(fmt, s [, pos])
```

**Example:**
```lua
local packed = string.pack("ii", 10, 20)
local a, b = string.unpack("ii", packed)
print(a, b)  -- 10  20
```
</Accordion>

<Accordion title="string.packsize">
Returns the size of a string created by `string.pack`.

```lua
string.packsize(fmt)
```

**Example:**
```lua
print(string.packsize("ii"))   -- 8
print(string.packsize("ddd"))  -- 24
```
</Accordion>

## Advanced Example

```lua
-- Parse CSV line
function parseCSV(line)
  local fields = {}
  for field in string.gmatch(line, '([^,]+)') do
    table.insert(fields, field)
  end
  return fields
end

local line = "John,Doe,30,Engineer"
local fields = parseCSV(line)
for i, field in ipairs(fields) do
  print(i, field)
end

-- Word wrap
function wordWrap(text, width)
  local lines = {}
  local line = ""
  
  for word in string.gmatch(text, "%S+") do
    if #line + #word + 1 > width then
      table.insert(lines, line)
      line = word
    else
      if #line > 0 then
        line = line .. " " .. word
      else
        line = word
      end
    end
  end
  
  if #line > 0 then
    table.insert(lines, line)
  end
  
  return table.concat(lines, "\n")
end

local text = "The quick brown fox jumps over the lazy dog"
print(wordWrap(text, 20))
```
